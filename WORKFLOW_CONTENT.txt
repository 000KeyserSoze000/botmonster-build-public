name: 'Tauri Release'

on:
  push:
    tags:
      - 'v*.*.*'

jobs:
  release:
    permissions:
      contents: write
      
    strategy:
      fail-fast: false
      matrix:
        platform: [macos-latest, ubuntu-22.04, windows-latest]
    runs-on: ${{ matrix.platform }}

    steps:
      - name: Checkout repository
        uses: actions/checkout@v4

      - name: Install Rust stable
        uses: dtolnay/rust-toolchain@stable

      - name: Setup Node.js
        uses: actions/setup-node@v4
        with:
          node-version: 20

      - name: Install pnpm
        uses: pnpm/action-setup@v3
        with:
          version: 8
          run_install: false # Nous gérons l'installation manuellement

      - name: Install Ubuntu dependencies
        if: matrix.platform == 'ubuntu-22.04'
        run: |
          sudo apt-get update
          sudo apt-get install -y \
            libwebkit2gtk-4.0-dev \
            libappindicator3-dev \
            librsvg2-dev \
            patchelf \
            libgtk-3-dev

      - name: Install Frontend Dependencies (Defensive)
        shell: bash
        run: |
          echo "--- Starting Defensive Dependency Installation ---"
          
          # Étape 1: Forcer l'utilisation du registre npm officiel (bonne pratique en CI)
          pnpm config set registry https://registry.npmjs.org/
          
          # Étape 2: Purger le cache pnpm. Le "|| true" assure que le job ne plante pas si le cache est déjà vide.
          echo "Pruning pnpm store..."
          pnpm store prune || true
          
          # Étape 3: Supprimer les artefacts potentiellement conflictuels des builds précédents
          echo "Cleaning up old artifacts..."
          rm -f pnpm-lock.yaml
          rm -rf node_modules
          
          # Étape 4: Installer les dépendances. pnpm va générer un nouveau lockfile à partir de package.json.
          # L'option --no-frozen-lockfile est explicite pour s'assurer qu'un nouveau lockfile est créé si nécessaire.
          echo "Installing dependencies and generating a fresh lockfile..."
          pnpm install --no-frozen-lockfile
          
          echo "--- Defensive Installation Complete ---"

      - name: Build and Release Tauri App
        uses: tauri-apps/tauri-action@v0
        env:
          GITHUB_TOKEN: ${{ secrets.GITHUB_TOKEN }}
        with:
          tagName: ${{ github.ref_name }}
          releaseName: "BotMonster ${{ github.ref_name }}"
          releaseBody: "New release of the BotMonster AI Trading Terminal."
          releaseDraft: false
          prerelease: false
