name: 'Tauri Release'

on:
  push:
    tags:
      - 'v*.*.*'

jobs:
  release:
    permissions:
      contents: write # This is required to create a release
    strategy:
      fail-fast: false
      matrix:
        platform: [macos-latest, ubuntu-22.04, windows-latest]
    runs-on: ${{ matrix.platform }}

    steps:
      - name: Checkout repository
        uses: actions/checkout@v4

      - name: Setup Node.js
        uses: actions/setup-node@v3
        with:
          node-version: 20

      - name: Install Rust stable
        uses: dtolnay/rust-toolchain@stable

      # This is required for the Tauri build on Ubuntu
      - name: Install dependencies (Ubuntu only)
        if: matrix.platform == 'ubuntu-22.04'
        run: |
          sudo apt-get update
          sudo apt-get install -y libwebkit2gtk-4.0-dev libappindicator3-dev librsvg2-dev patchelf libgtk-3-dev

      - name: Clean Install Frontend Dependencies
        run: |
          npm cache clean --force
          npm ci --no-cache

      - name: Build and Release Tauri App
        uses: tauri-apps/tauri-action@v0
        env:
          GITHUB_TOKEN: ${{ secrets.GITHUB_TOKEN }}
        with:
          # tauri-action automatically reads the version from tauri.conf.json
          # and creates a release based on the git tag.
          tagName: ${{ github.ref_name }}
          releaseName: "BotMonster v__VERSION__"
          releaseBody: "New release of the BotMonster AI Trading Terminal. Download the appropriate installer for your system below."
          releaseDraft: false
          prerelease: false
